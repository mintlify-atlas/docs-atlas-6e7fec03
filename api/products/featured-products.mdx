---
title: "Get Featured Products"
api: "GET /api/productos/destacados"
description: "Retrieve all products marked as featured"
---

## Authentication

This endpoint is publicly accessible and does not require authentication.

## Response

Returns an array of featured product objects.

<ResponseField name="productos" type="array">
  Array of featured product objects
  
  <Expandable title="Product Object">
    <ResponseField name="producto_id" type="long">
      Unique identifier for the product
    </ResponseField>
    
    <ResponseField name="sku" type="string">
      Stock Keeping Unit
    </ResponseField>
    
    <ResponseField name="nombre" type="string">
      Product name
    </ResponseField>
    
    <ResponseField name="descripcion" type="string">
      Product description
    </ResponseField>
    
    <ResponseField name="precioCantidad" type="decimal">
      Price amount
    </ResponseField>
    
    <ResponseField name="precioMoneda" type="string">
      Currency code
    </ResponseField>
    
    <ResponseField name="dimensionesAlto" type="double">
      Height dimension
    </ResponseField>
    
    <ResponseField name="dimensionesAncho" type="double">
      Width dimension
    </ResponseField>
    
    <ResponseField name="dimensionesProfundo" type="double">
      Depth dimension
    </ResponseField>
    
    <ResponseField name="es_destacado" type="boolean">
      Always true for featured products
    </ResponseField>
    
    <ResponseField name="stock" type="integer">
      Available quantity
    </ResponseField>
    
    <ResponseField name="imagen_url" type="string">
      Product image URL
    </ResponseField>
    
    <ResponseField name="categoria" type="object">
      Category information
      
      <Expandable title="Category Object">
        <ResponseField name="categoria_id" type="long">
          Category unique identifier
        </ResponseField>
        
        <ResponseField name="nombre" type="string">
          Category name
        </ResponseField>
        
        <ResponseField name="slug" type="string">
          URL-friendly category identifier
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl --location 'https://api.iquea.com/api/productos/destacados'
```

```javascript JavaScript
fetch('https://api.iquea.com/api/productos/destacados')
  .then(response => response.json())
  .then(data => console.log(data));
```

```python Python
import requests

response = requests.get('https://api.iquea.com/api/productos/destacados')
featured_products = response.json()
```
</RequestExample>

</ResponseField>
