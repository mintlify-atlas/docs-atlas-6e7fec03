---
title: "List Products"
api: "GET /api/productos"
description: "Retrieve a list of all products in the catalog"
---

## Authentication

This endpoint is publicly accessible and does not require authentication.

## Response

Returns an array of product objects with complete details.

<ResponseField name="productos" type="array">
  Array of product objects
  
  <Expandable title="Product Object">
    <ResponseField name="producto_id" type="long">
      Unique identifier for the product
    </ResponseField>
    
    <ResponseField name="sku" type="string">
      Stock Keeping Unit - unique product code
    </ResponseField>
    
    <ResponseField name="nombre" type="string">
      Product name
    </ResponseField>
    
    <ResponseField name="descripcion" type="string">
      Detailed product description
    </ResponseField>
    
    <ResponseField name="precioCantidad" type="decimal">
      Price amount
    </ResponseField>
    
    <ResponseField name="precioMoneda" type="string">
      Currency code (e.g., "USD", "EUR")
    </ResponseField>
    
    <ResponseField name="dimensionesAlto" type="double">
      Product height dimension
    </ResponseField>
    
    <ResponseField name="dimensionesAncho" type="double">
      Product width dimension
    </ResponseField>
    
    <ResponseField name="dimensionesProfundo" type="double">
      Product depth dimension
    </ResponseField>
    
    <ResponseField name="es_destacado" type="boolean">
      Whether the product is featured
    </ResponseField>
    
    <ResponseField name="stock" type="integer">
      Available quantity in inventory
    </ResponseField>
    
    <ResponseField name="imagen_url" type="string">
      URL to product image
    </ResponseField>
    
    <ResponseField name="categoria" type="object">
      Category information
      
      <Expandable title="Category Object">
        <ResponseField name="categoria_id" type="long">
          Category unique identifier
        </ResponseField>
        
        <ResponseField name="nombre" type="string">
          Category name
        </ResponseField>
        
        <ResponseField name="slug" type="string">
          URL-friendly category identifier
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl --location 'https://api.iquea.com/api/productos'
```

```javascript JavaScript
fetch('https://api.iquea.com/api/productos')
  .then(response => response.json())
  .then(data => console.log(data));
```

```python Python
import requests

response = requests.get('https://api.iquea.com/api/productos')
products = response.json()
```
</RequestExample>
